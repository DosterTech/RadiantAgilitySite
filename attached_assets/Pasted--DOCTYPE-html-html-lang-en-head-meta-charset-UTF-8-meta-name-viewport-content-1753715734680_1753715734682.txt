    <script>
        function calculateScore() {
            const form = document.getElementById('assessmentForm');
            const formData = new FormData(form);
            
            let totalScore = 0;
            let answeredQuestions = 0;
            
            // Check if all questions are answered
            for (let i = 1; i <= 8; i++) {
                const answer = formData.get(`q${i}`);
                if (answer) {
                    totalScore += parseInt(answer);
                    answeredQuestions++;
                }
            }
            
            if (answeredQuestions < 8) {
                alert('Please answer all questions to get your personalized assessment.');
                return;
            }
            
            displayResults(totalScore);
        }
        
        function displayResults(score) {
            const resultsDiv = document.getElementById('results');
            const scoreDisplay = document.getElementById('scoreDisplay');
            const resultTitle = document.getElementById('resultTitle');
            const resultDescription = document.getElementById('resultDescription');

            let level, title, description, recommendations;
            
            if (score <= 12) {
                level = 'level-1';
                title = 'Activity-Focused Team';
                description = 'Your team is working hard but may be trapped in the "feature factory" mindset. There\'s significant opportunity to shift toward value-driven delivery.';
