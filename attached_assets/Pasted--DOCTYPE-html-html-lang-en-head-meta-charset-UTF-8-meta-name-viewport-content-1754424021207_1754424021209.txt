        function calculateWIP() {
            // Get input values
            const teamSize = parseInt(document.getElementById('teamSize').value);
            const sprintLength = parseInt(document.getElementById('sprintLength').value);
            const avgStorySize = parseInt(document.getElementById('avgStorySize').value);
            const velocity = parseInt(document.getElementById('velocity').value);
            const blockerRate = parseInt(document.getElementById('blockerRate').value);
            const multitasking = document.getElementById('multitasking').value;
            
            // Validate inputs
            if (!teamSize || !sprintLength || !avgStorySize || !velocity || blockerRate === null || !multitasking) {
                alert('Please fill in all fields to calculate your WIP limits.');
                return;
            }
            
            // Calculate base WIP limit
            const storiesPerSprint = Math.round(velocity / avgStorySize);
            const baseWIP = Math.max(2, Math.round(storiesPerSprint * 0.6));
            
            // Adjust for multitasking level
            let multitaskingMultiplier = 1;
            switch(multitasking) {
                case 'low': multitaskingMultiplier = 0.8; break;
                case 'medium': multitaskingMultiplier = 1.0; break;
                case 'high': multitaskingMultiplier = 1.3; break;
            }
            
            // Adjust for blocker rate
            const blockerMultiplier = 1 + (blockerRate / 100 * 0.5);
            
            // Calculate final WIP limit
            const wipLimit = Math.max(2, Math.round(baseWIP * multitaskingMultiplier * blockerMultiplier));
            const storiesPerPerson = Math.round((wipLimit / teamSize) * 10) / 10;
            const blockerBuffer = Math.round(wipLimit * (blockerRate / 100));
            const cycleTime = Math.round((wipLimit / (velocity / (sprintLength * 5))) * 10) / 10; // days
            
            // Display results
            document.getElementById('wipLimit').textContent = wipLimit;
            document.getElementById('storiesPerPerson').textContent = storiesPerPerson;
            document.getElementById('blockerBuffer').textContent = blockerBuffer + ' stories';
            document.getElementById('cycleTime').textContent = cycleTime + ' days';
            
            // Show results section
            document.getElementById('results').style.display = 'block';
            
            // Generate recommendations
            generateRecommendations(teamSize, wipLimit, blockerRate, multitasking, storiesPerPerson);
            
            // Scroll to results
            document.getElementById('results').scrollIntoView({ behavior: 'smooth' });
        }
        
        function generateRecommendations(teamSize, wipLimit, blockerRate, multitasking, storiesPerPerson) {
            const recommendations = [];
            
            if (storiesPerPerson > 2) {
                recommendations.push("Consider reducing WIP further - each person should ideally work on 1-2 stories maximum.");
            }
            
            if (blockerRate > 20) {
                recommendations.push("High blocker rate detected. Focus on identifying and removing impediments before optimizing WIP.");
            }
            
            if (multitasking === 'high') {
                recommendations.push("High multitasking level may hurt flow. Try encouraging team members to focus on fewer items.");
            }
            
            if (wipLimit > teamSize * 2) {
                recommendations.push("WIP limit seems high relative to team size. Start lower and gradually increase if needed.");
            }
            
            if (teamSize > 8) {
                recommendations.push("Large team detected. Consider splitting into smaller sub-teams with separate WIP limits.");
            }
            
            // Always include these general recommendations
            recommendations.push("Start with these limits for 2-3 sprints, then adjust based on actual flow metrics.");
            recommendations.push("Make WIP limits visible on your board and enforce them consistently.");
            recommendations.push("Track cycle time and throughput to measure the impact of your WIP limits.");
            
            // Display recommendations
            const recommendationsList = document.getElementById('recommendationsList');
            recommendationsList.innerHTML = '';
            
            recommendations.forEach((rec, index) => {
                const item = document.createElement('div');
                item.className = 'recommendation-item';
                item.innerHTML = `
                    <div class="recommendation-icon">${index + 1}</div>
                    <div>${rec}</div>
                `;
                recommendationsList.appendChild(item);
            });
            
            document.getElementById('recommendations').style.display = 'block';
        }
