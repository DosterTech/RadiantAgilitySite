    <script>
        function addStage() {
            const tbody = document.getElementById('journeyBody');
            const newRow = document.createElement('tr');
            newRow.innerHTML = `
                <td class="stage-cell"><div contenteditable="true" class="editable">New Stage</div></td>
                <td><div contenteditable="true" class="editable">Enter actions...</div></td>
                <td><div contenteditable="true" class="editable">Enter thoughts...</div></td>
                <td><span class="emotion-score emotion-neutral">üòê Neutral</span></td>
                <td><div contenteditable="true" class="editable">Enter touchpoints...</div></td>
                <td><div contenteditable="true" class="editable">Enter pain points...</div></td>
                <td><div contenteditable="true" class="editable">Enter opportunities...</div></td>
            `;
            tbody.appendChild(newRow);
            attachEmotionListeners();
        }

        function attachEmotionListeners() {
            document.querySelectorAll('.emotion-score').forEach(span => {
                span.style.cursor = 'pointer';
                span.onclick = function() {
                    cycleEmotion(this);
                };
            });
        }

        function cycleEmotion(element) {
            const emotions = [
                { class: 'emotion-positive', text: 'üòä Happy' },
                { class: 'emotion-positive', text: 'üéâ Excited' },
                { class: 'emotion-neutral', text: 'üòê Neutral' },
                { class: 'emotion-neutral', text: 'üòï Confused' },
                { class: 'emotion-negative', text: 'üòî Frustrated' },
                { class: 'emotion-negative', text: 'üò° Angry' }
            ];
            
            const currentText = element.textContent;
            const currentIndex = emotions.findIndex(e => e.text === currentText);
            const nextIndex = (currentIndex + 1) % emotions.length;
            const nextEmotion = emotions[nextIndex];
            
            element.className = `emotion-score ${nextEmotion.class}`;
            element.textContent = nextEmotion.text;
        }

        function updateJourneyInfo() {
            const persona = document.getElementById('persona').value;
            const scenario = document.getElementById('scenario').value;
            const type = document.getElementById('journey-type').value;
            
            alert(`Journey updated:\nPersona: ${persona}\nScenario: ${scenario}\nType: ${type}`);
        }

        function downloadHTML() {
            const content = document.documentElement.outerHTML;
            const blob = new Blob([content], { type: 'text/html' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = 'journey-map-template.html';
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            URL.revokeObjectURL(url);
        }

        function exportData() {
            const data = {
                persona: document.getElementById('persona').value,
                scenario: document.getElementById('scenario').value,
                type: document.getElementById('journey-type').value,
                stages: []
            };
            
            document.querySelectorAll('#journeyBody tr').forEach(row => {
                const cells = row.querySelectorAll('td');
                data.stages.push({
                    stage: cells[0].textContent.trim(),
                    actions: cells[1].textContent.trim(),
                    thoughts: cells[2].textContent.trim(),
                    emotions: cells[3].textContent.trim(),
                    touchpoints: cells[4].textContent.trim(),
                    painPoints: cells[5].textContent.trim(),
                    opportunities: cells[6].textContent.trim()
                });
            });
            
            const jsonStr = JSON.stringify(data, null, 2);
            const blob = new Blob([jsonStr], { type: 'application/json' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = 'journey-map-data.json';
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            URL.revokeObjectURL(url);
        }

        // Initialize emotion listeners
        attachEmotionListeners();

        // Auto-save to localStorage
        setInterval(() => {
            const content = document.getElementById('journeyBody').innerHTML;
            localStorage.setItem('journeyMapContent', content);
        }, 5000);

        // Load saved content
        window.onload = function() {
            const saved = localStorage.getItem('journeyMapContent');
            if (saved && confirm('Load previously saved journey map?')) {
                document.getElementById('journeyBody').innerHTML = saved;
                attachEmotionListeners();
            }
        };
    </script>
