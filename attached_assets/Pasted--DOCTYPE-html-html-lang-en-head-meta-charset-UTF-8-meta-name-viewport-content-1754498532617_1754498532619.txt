    <script>
        let selectedType = 'emotion';

        function selectType(btn, type) {
            document.querySelectorAll('.type-btn').forEach(b => b.classList.remove('active'));
            btn.classList.add('active');
            selectedType = type;
        }

        function generatePrompt() {
            const context = document.getElementById('journey-context').value;
            const persona = document.getElementById('persona').value;
            const touchpoints = document.getElementById('touchpoints').value;
            const focus = document.getElementById('specific-focus').value;

            let promptTemplate = `You are an expert UX researcher and customer experience analyst. Analyze the following customer journey for emotional states and friction points.

JOURNEY CONTEXT:
${context || '[Describe your journey here]'}

TARGET PERSONA:
${persona || '[Define your target persona]'}

KEY TOUCHPOINTS:
${touchpoints || '[List your touchpoints]'}

${focus ? `SPECIFIC FOCUS AREAS:\n${focus}\n` : ''}
ANALYSIS REQUIRED:
`;

            if (selectedType === 'emotion') {
                promptTemplate += `
1. EMOTION MAPPING:
   - Identify the emotional state at each touchpoint
   - Rate emotional intensity (1-10)
   - Highlight emotional peaks and valleys
   - Suggest emotion-driven design improvements

2. EMOTIONAL DRIVERS:
   - What triggers positive emotions?
   - What causes negative emotional responses?
   - Where do emotions shift dramatically?

3. EMPATHY INSIGHTS:
   - Key customer thoughts/feelings
   - Unspoken needs and desires
   - Moments of delight vs. frustration`;
            } else if (selectedType === 'friction') {
                promptTemplate += `
1. FRICTION ANALYSIS:
   - List all friction points in order of severity
   - Categorize friction types (Technical, Cognitive, Emotional)
   - Estimate impact on conversion/satisfaction

2. ROOT CAUSES:
   - Why does each friction point exist?
   - Is it systemic or isolated?
   - Quick fixes vs. strategic changes needed

3. FRICTION HEAT MAP:
   - Which touchpoints have highest friction?
   - Cumulative friction score
   - Breaking points where users abandon`;
            } else if (selectedType === 'both') {
                promptTemplate += `
1. COMBINED EMOTION & FRICTION ANALYSIS:
   - Map emotions at each touchpoint (😊😐😔😡)
   - Identify friction severity (Low/Medium/High/Critical)
   - Show correlation between friction and emotional response

2. JOURNEY SCORING:
   - Overall emotional score: /10
   - Total friction score: /10
   - Net experience score: /10

3. PRIORITY MATRIX:
   - High Impact + Quick Win improvements
   - Strategic long-term optimizations
   - Critical fixes needed immediately`;
            } else {
                promptTemplate += `
1. OPPORTUNITY DISCOVERY:
   - Top 5 improvement opportunities ranked by impact
   - Innovation possibilities at each touchpoint
   - Competitive differentiation opportunities

2. VALUE CREATION:
   - How to exceed expectations
   - Surprise and delight moments
   - Personalization opportunities

3. IMPLEMENTATION ROADMAP:
   - Phase 1: Quick wins (1-2 weeks)
   - Phase 2: Medium efforts (1-3 months)
   - Phase 3: Transformational changes (3-6 months)`;
            }

            promptTemplate += `

OUTPUT FORMAT:
- Use clear headers and bullet points
- Include specific examples and scenarios
- Provide actionable recommendations
- Rate confidence level for each insight (High/Medium/Low)
- Suggest metrics to track improvements

Please provide a comprehensive analysis that will help improve this customer journey.`;

            document.getElementById('prompt-text').textContent = promptTemplate;
            
            // Animate the output
            const output = document.querySelector('.prompt-output');
            output.style.animation = 'none';
            setTimeout(() => {
                output.style.animation = 'fadeInUp 0.6s ease-out';
            }, 10);
        }

        function copyPrompt() {
            const promptText = document.getElementById('prompt-text').textContent;
            navigator.clipboard.writeText(promptText).then(() => {
                const btn = document.querySelector('.copy-btn');
                btn.textContent = '✅ Copied!';
                btn.classList.add('copied');
                setTimeout(() => {
                    btn.textContent = '📋 Copy';
                    btn.classList.remove('copied');
                }, 2000);
            });
        }

        function resetForm() {
            document.getElementById('journey-context').value = '';
            document.getElementById('persona').value = '';
            document.getElementById('touchpoints').value = '';
            document.getElementById('specific-focus').value = '';
            document.getElementById('prompt-text').textContent = 'Your AI prompt will appear here after you click "Generate Prompt"...';
            
            document.querySelectorAll('.type-btn').forEach(b => b.classList.remove('active'));
            document.querySelector('.type-btn').classList.add('active');
            selectedType = 'emotion';
        }
